// ── Leaflet карта (ленивая инициализация) ──
let map = null, trackLayer = null, markerStart = null, markerEnd = null;

function initMap(lat, lon, track) {
  if (!map) {
    map = L.map('map', { zoomControl: true }).setView([lat, lon], 5);
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '© OpenStreetMap contributors', maxZoom: 18
    }).addTo(map);
  }
  if (trackLayer) map.removeLayer(trackLayer);
  if (markerStart) map.removeLayer(markerStart);
  if (markerEnd) map.removeLayer(markerEnd);

  const latlngs = track.map(p => [p.lat, p.lon]);

  // Градиент цвета: синий → красный по времени
  for (let i = 0; i < latlngs.length - 1; i++) {
    const t = i / (latlngs.length - 1);
    const r = Math.round(t * 220);
    const b = Math.round((1-t) * 200 + 55);
    L.polyline([latlngs[i], latlngs[i+1]], {
      color: `rgb(${r},80,${b})`, weight: 3, opacity: .85
    }).addTo(map);
  }

  const iconStart = L.divIcon({ html: '<div style="background:#4f98a3;width:14px;height:14px;border-radius:50%;border:2px solid #fff;"></div>', iconSize:[14,14], iconAnchor:[7,7] });
  const iconEnd   = L.divIcon({ html: '<div style="background:#e05a5a;width:14px;height:14px;border-radius:50%;border:2px solid #fff;"></div>', iconSize:[14,14], iconAnchor:[7,7] });
  markerStart = L.marker(latlngs[0], {icon: iconStart}).bindPopup('Начало').addTo(map);
  markerEnd   = L.marker(latlngs[latlngs.length-1], {icon: iconEnd}).bindPopup('Конец прогноза').addTo(map);

  // Часовые маркеры каждые 24 ч
  track.filter(p => p.h % 24 === 0 && p.h > 0).forEach(p => {
    L.circleMarker([p.lat, p.lon], {radius:4, color:'#d4a843', fillColor:'#d4a843', fillOpacity:.8})
     .bindPopup(`${p.h} ч`).addTo(map);
  });

  map.fitBounds(L.latLngBounds(latlngs).pad(.2));
}

// ── Загрузка файла ──
const fileInput = document.getElementById('file-input');
const dropZone  = document.getElementById('drop-zone');
const runBtn    = document.getElementById('run-btn');
let selectedFile = null;

fileInput.addEventListener('change', e => setFile(e.target.files[0]));
dropZone.addEventListener('dragover', e => { e.preventDefault(); dropZone.classList.add('drag'); });
dropZone.addEventListener('dragleave', () => dropZone.classList.remove('drag'));
dropZone.addEventListener('drop', e => { e.preventDefault(); dropZone.classList.remove('drag'); setFile(e.dataTransfer.files[0]); });

function setFile(file) {
  if (!file) return;
  selectedFile = file;
  const reader = new FileReader();
  reader.onload = ev => {
    document.getElementById('preview-img').src = ev.target.result;
    document.getElementById('preview-wrap').style.display = 'block';
    dropZone.style.display = 'none';
    runBtn.disabled = false;
  };
  reader.readAsDataURL(file);
}

function clearImage() {
  selectedFile = null;
  document.getElementById('preview-wrap').style.display = 'none';
  dropZone.style.display = 'block';
  fileInput.value = '';
  runBtn.disabled = true;
}

// ── Запрос ──
async function runPredict() {
  if (!selectedFile) return;
  document.getElementById('btn-text').style.display = 'none';
  document.getElementById('spinner').style.display = 'block';
  runBtn.disabled = true;

  const fd = new FormData();
  fd.append('image', selectedFile);
  fd.append('lat',    document.getElementById('lat').value);
  fd.append('lon',    document.getElementById('lon').value);
  fd.append('wind_u', document.getElementById('wind_u').value);
  fd.append('wind_v', document.getElementById('wind_v').value);
  fd.append('curr_u', document.getElementById('curr_u').value);
  fd.append('curr_v', document.getElementById('curr_v').value);
  fd.append('hours',  document.getElementById('hours').value);

  try {
    const res = await fetch('/predict', { method: 'POST', body: fd });
    const d = await res.json();
    if (d.error) { alert(d.error); return; }
    renderResults(d);
  } catch(e) {
    alert('Ошибка запроса: ' + e.message);
  } finally {
    document.getElementById('btn-text').style.display = 'block';
    document.getElementById('spinner').style.display = 'none';
    runBtn.disabled = false;
  }
}

function renderResults(d) {
  const el = document.getElementById('results');
  const isIce = d.is_iceberg;
  const pct = Math.round(d.prob * 100);
  const badgeCls = isIce ? 'badge-iceberg' : 'badge-ship';

  // ── Детекции: текст строк ──
  let detHTML = '';
  if (d.detections.length === 0) {
    detHTML = '<div class="no-det">Объекты не обнаружены</div>';
  } else {
    detHTML = '<div class="det-list">' + d.detections.map(det => {
      const tag = det.cls_name.toLowerCase().includes('ice') ? 'tag-ice' : 'tag-ship';
      return `<div class="det-item">
        <div>
          <span class="tag ${tag}">${det.cls_name}</span>
          <div class="coords">bbox: [${det.xyxy.join(', ')}]</div>
        </div>
        <span class="conf">${(det.conf*100).toFixed(1)}%</span>
      </div>`;
    }).join('') + '</div>';
  }

  // ── Дрейф ──
  let driftHTML = '';
  if (isIce && d.drift_track.length > 0) {
    const last = d.drift_track[d.drift_track.length - 1];
    const start = d.drift_track[0];
    const dLat = (last.lat - start.lat).toFixed(3);
    const dLon = (last.lon - start.lon).toFixed(3);
    driftHTML = `
      <div class="card">
        <div class="card-header">🗺️ Прогноз траектории дрейфа</div>
        <div class="card-body">
          <div id="map"></div>
          <div class="drift-stats">
            <div class="stat-box"><div class="val">${d.drift_km} км</div><div class="lbl">Смещение</div></div>
            <div class="stat-box"><div class="val">${last.h} ч</div><div class="lbl">Горизонт</div></div>
            <div class="stat-box"><div class="val">${last.lat.toFixed(2)}°, ${last.lon.toFixed(2)}°</div><div class="lbl">Конечная точка</div></div>
          </div>
        </div>
      </div>`;
  } else if (!isIce) {
    driftHTML = `<div class="card"><div class="card-body" style="color:var(--muted);text-align:center;padding:20px">
      Объект классифицирован как корабль — прогноз дрейфа не применяется
    </div></div>`;
  }

  el.innerHTML = `
    <!-- Классификация -->
    <div class="card">
      <div class="card-header">🧊 Классификация объекта</div>
      <div class="card-body">
        <div class="verdict">
          <div class="verdict-badge ${badgeCls}">${d.label}</div>
          <div class="verdict-meta">
            <strong>${pct}%</strong> вероятность<br>
            <span>Порог: 50%</span>
          </div>
        </div>
        <div class="prob-bar" style="margin-top:14px">
          <div class="prob-fill" style="width:${pct}%"></div>
        </div>
      </div>
    </div>

    <!-- Детекция -->
    <div class="card">
      <div class="card-header">📡 Детекция YOLO (bbox)</div>
      <div class="card-body">${detHTML}</div>
    </div>

    ${driftHTML}
  `;

  // Инициализируем карту после того, как div#map появился в DOM
  if (isIce && d.drift_track.length > 0) {
    setTimeout(() => {
      initMap(d.drift_track[0].lat, d.drift_track[0].lon, d.drift_track);
    }, 50);
  }
}
